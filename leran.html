Basic Report Structure

JavaScript Code

// 1. Define report data
const reportData = [
    { id: 1, name: 'Product A', sales: 1500, category: 'Electronics' },
    { id: 2, name: 'Product B', sales: 2300, category: 'Clothing' },
    { id: 3, name: 'Product C', sales: 1800, category: 'Electronics' }
];

// 2. Calculate totals
function generateReport(data) {
    const totalSales = data.reduce((sum, item) => sum + item.sales, 0);
    const averageSales = totalSales / data.length;
    
    // 3. Return report object
    return {
        items: data,
        summary: {
            totalItems: data.length,
            totalSales: totalSales,
            averageSales: averageSales.toFixed(2)
        }
    };
}

// 4. Generate and display report
const report = generateReport(reportData);
console.log('Report Summary:', report.summary);

const salesData = [

    { date: '2023-01-01', product: 'Phone', amount: 800, region: 'South' },
    { date: '2023-01-02', product: 'Laptop', amount: 1500, region: 'North' },
    { date: '2023-01-02', product: 'Tablet', amount: 600, region: 'East' }
];

// 1. Filtering
const northSales = salesData.filter(item => item.region === 'North');

// 2. Grouping by property
function groupBy(array, key) {
    return array.reduce((result, item) => {
        const groupKey = item[key];
        if (!result[groupKey]) result[groupKey] = [];
        result[groupKey].push(item);
        return result;
    }, {});
}

// 3. Aggregation
function aggregateSales(data) {
    const byProduct = groupBy(data, 'product');
    const summary = {};
    
    for (const product in byProduct) {
        const items = byProduct[product];
        summary[product] = {
            totalSales: items.reduce((sum, item) => sum + item.amount, 0),
            count: items.length,
            average: items.reduce((sum, item) => sum + item.amount, 0) / items.length
        };
    }
    
    return summary;
}

// Employee data
const employees = [

    { id: 2, name: 'Jane Smith', department: 'Marketing', salary: 60000, joinDate: '2019-03-22' },
    { id: 3, name: 'Bob Johnson', department: 'Sales', salary: 55000, joinDate: '2021-06-10' }
];

// Function to create dynamic table
function createReportTable(data, containerId) {
    const container = document.getElementById(containerId);
    
    // Create table element
    const table = document.createElement('table');
    table.className = 'report-table';
    
    // Create table header
    const thead = document.createElement('thead');
    const headers = Object.keys(data[0]);
    let headerRow = '';
    headers.forEach(header => {
        headerRow += `${header.charAt(0).toUpperCase() + header.slice(1)}`;
    });
    headerRow += '';
    thead.innerHTML = headerRow;
    
    // Create table body
    const tbody = document.createElement('tbody');
    data.forEach(item => {
        let row = '';
        headers.forEach(header => {
            row += `${item[header]}`;
        });
        row += '';
        tbody.innerHTML += row;
    });
    
    // Assemble table
    table.appendChild(thead);
    table.appendChild(tbody);
    container.innerHTML = '';
    container.appendChild(table);
}

// Function to add sorting
function addTableSorting(tableId) {
    const table = document.getElementById(tableId);
    const headers = table.querySelectorAll('th');
    
    headers.forEach((header, index) => {
        header.style.cursor = 'pointer';
        header.addEventListener('click', () => {
            sortTable(table, index);
        });
    });
}

Using Chart.js for Reports

JavaScript Code

// Sample sales data for charts
const monthlySales = {
    labels: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun'],
    datasets: [{
        label: 'Sales 2023',
        data: [12000, 19000, 15000, 25000, 22000, 30000],
        backgroundColor: 'rgba(54, 162, 235, 0.5)',
        borderColor: 'rgba(54, 162, 235, 1)',
        borderWidth: 2
    }]
};

// Create bar chart
function createBarChart() {
    const ctx = document.getElementById('salesChart').getContext('2d');
    return new Chart(ctx, {
        type: 'bar',
        data: monthlySales,
        options: {
            responsive: true,
            plugins: {
                title: {
                    display: true,
                    text: 'Monthly Sales Report 2023'
                },
                tooltip: {
                    callbacks: {
                        label: function(context) {
                            return `Sales: $${context.raw.toLocaleString()}`;
                        }
                    }
                }
            },
            scales: {
                y: {
                    beginAtZero: true,
                    ticks: {
                        callback: function(value) {
                            return '$' + value.toLocaleString();
                        }
                    }
                }
            }
        }
    });
}

PDF Report Generation

Using jsPDF for PDF Reports

JavaScript Code

// Generate PDF Report
function generatePDFReport() {
    // Create new PDF document
    const pdf = new jsPDF();
    
    // Add title
    pdf.setFontSize(20);
    pdf.setTextColor(44, 62, 80);
    pdf.text("Sales Report - Q1 2023", 105, 20, { align: 'center' });
    
    // Add date
    pdf.setFontSize(12);
    pdf.setTextColor(128, 128, 128);
    pdf.text(`Generated: ${new Date().toLocaleDateString()}`, 105, 30, { align: 'center' });
    
    // Add table
    pdf.autoTable({
        startY: 40,
        head: [['Product', 'Q1 Sales', 'Growth %', 'Region']],
        body: [
            ['Laptops', '$125,000', '+12%', 'North America'],
            ['Phones', '$89,500', '+8%', 'Europe'],
            ['Tablets', '$67,300', '+15%', 'Asia'],
            ['Accessories', '$45,200', '+5%', 'Global']
        ],
        theme: 'striped',
        headStyles: { fillColor: [52, 152, 219] },
        margin: { left: 20, right: 20 }
    });
    
    // Add summary
    const finalY = pdf.lastAutoTable.finalY + 15;
    pdf.setFontSize(14);
    pdf.setTextColor(44, 62, 80);
    pdf.text("Summary", 20, finalY);
    
    pdf.setFontSize(12);
    pdf.setTextColor(0, 0, 0);
    const summary = [
        '• Total Q1 Sales: $327,000',
        '• Average Growth: 10%',
        '• Top Performing: Tablets (+15%)',
        '• Regions: All showing positive growth'
    ];
    
    summary.forEach((item, index) => {
        pdf.text(item, 25, finalY + 10 + (index * 7));
    });
    
    // Save the PDF
    pdf.save('sales-report-q1-2023.pdf');
}

// Generate from HTML
function generateFromHTML() {
    const element = document.getElementById('reportContent');
    html2canvas(element).then(canvas => {
        const imgData = canvas.toDataURL('image/png');
        const pdf = new jsPDF();
        pdf.addImage(imgData, 'PNG', 10, 10, 190, 0);
        pdf.save('html-report.pdf');
    });
}
